# views.py
def get_context_data():
    context["questions"]  = Question.objects.prefetch_related('choices')
    return context

# templates/.html
{% for question in questions %}
    <b>{{ question.question_text }}</b>
    <ul>
        {% for choice in question.choices.all %}
            <li>{{ choice.choice_text }}</li>
        {% endfor %}
    </ul>
{% endfor %}

import sys

A, B = sys.stdin.readline().split()

NA = int(A[::-1])
NB = int(B[::-1])

if NA > NB :
    print(NA)
else: 
    print(NB)
    
# 문자열 슬라이싱을 잘 활용하면 아주아주 좋다.

import sys

N = int(sys.stdin.readline().rstrip())

for i in range(N):
    sent = sys.stdin.readline().rstrip()
    for j in range(1, len(sent)):
        if sent.find(sent[j-1]) > sent.find(sent[j]):
            N -= 1
            break
print(N)

# find함수도 잘 알아두면 좋다.

    N, K = map(int, input().split())
    score = list(map(int, input().split()))

    score.sort(reverse=True)

    result = 0
    for i in range(K):
        result += score[i]

    print('#{} {}'.format(T+1, result))

for T in range(int(input())):
    D, H, M = map(int, input().split())
    result = D*1440 + H*60 + M
    if result >= 16511:
        print(f'#{T+1} {result-16511}')

        print(f'#{T+1} -1')

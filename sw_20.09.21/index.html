for T in range(int(input())):
    A, B = input().split()
    dp = [[0 for i in range(len(A)+1)] for j in range(len(B)+1)]

    for i in range(1, len(B)+1):
        for j in range(1, len(A)+1):
            if B[i-1] == A[j-1]:
                dp[i][j] = dp[i-1][j-1] + 1
            else:
                dp[i][j] = max(dp[i][j-1], dp[i-1][j])
                            
    print(f'#{T+1} {dp[-1][-1]}')

def exeu(a, b):
    global x, y
    r = [a, b]
    s = [1, 0]
    t = [0, 1]

    while r[-1] != 0:
        q = int(r[-2] / r[-1])
        r.append(r[-2] - q * r[-1])
        s.append(s[-2] - q * s[-1])
        t.append(t[-2] - q * t[-1])
    x = s[-2]
    y = t[-2]

for T in range(int(input())):
    A, B = map(int, input().split())
    exeu(A, B)

    print(f'#{T+1} {x} {y}')
